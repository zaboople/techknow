# Loop thru some strings
for i in "1" "2" "3"; do  echo $i; done

# Loop on a number:
for (( i=1; i <= 105; i++ ))
do
  curl http://localhost:8080/ &
done

# Loop thru arguments:
for drive in "$@"
  do
  echo $drive
  done

for x in ${@}; do
    echo $x
done

# A better way to loop thru arrays:
for x in ${ray[*]}; do
  echo "ray $x"
done

# A better way to loop thru associative arrays (!)
for x in "${!ray[@]}"; do
  echo "ray $x"
done

# Okay while is good too:
while read line; do
  echo $line
done;
